  # Check if under 200ms (0.2s)
  if (( $(echo "$response_time < 0.2" | bc -l) )); then
    echo "‚úÖ PASS: Under 200ms target"
  else
    echo "‚ö†Ô∏è WARNING: Over 200ms target"
  fi
  
  echo "---"
